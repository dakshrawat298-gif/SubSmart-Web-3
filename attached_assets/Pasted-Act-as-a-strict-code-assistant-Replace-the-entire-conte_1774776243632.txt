  <script>
    const connectBtn = document.getElementById('connectWalletBtn');
    const walletAddressText = document.getElementById('walletAddress');
    const warningAlert = document.getElementById('walletWarning');
    const createPlanBtn = document.getElementById('createPlanBtn');
    const terminalBody = document.getElementById('terminalBody');
    const planAmountInput = document.getElementById('planAmount');
    
    let userAddress = null;
    let planCount = 0;

    function addLog(text, color = "text-green-400") {
      const p = document.createElement('p');
      p.className = color;
      p.innerText = `> ${text}`;
      terminalBody.appendChild(p);
      terminalBody.scrollTop = terminalBody.scrollHeight;
    }

    connectBtn.addEventListener('click', async () => {
      if (typeof window.ethereum !== 'undefined') {
        try {
          const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' });
          userAddress = accounts[0];
          const truncated = userAddress.substring(0, 6) + "..." + userAddress.substring(userAddress.length - 4);
          
          walletAddressText.innerText = truncated;
          connectBtn.classList.replace('bg-gray-900', 'bg-green-600');
          warningAlert.style.display = 'none';
          
          createPlanBtn.disabled = false;
          createPlanBtn.classList.remove('opacity-50', 'cursor-not-allowed');
          
          addLog(`Wallet connected: ${truncated}`, "text-white font-bold");
        } catch (error) {
          addLog("Connection rejected by user.", "text-red-400");
        }
      } else {
        alert("MetaMask is not installed!");
      }
    });

    createPlanBtn.addEventListener('click', async () => {
      if (!planAmountInput.value) {
        alert("Please enter an amount.");
        return;
      }

      createPlanBtn.disabled = true;
      createPlanBtn.innerText = "Please confirm in MetaMask...";
      addLog("Waiting for MetaMask confirmation...", "text-yellow-400");

      try {
        const provider = new ethers.providers.Web3Provider(window.ethereum);
        const signer = provider.getSigner();
        
        // This simulates a real smart contract interaction popup by sending a 0 value transaction to oneself
        const tx = await signer.sendTransaction({
          to: userAddress,
          value: ethers.utils.parseEther("0.0")
        });

        addLog(`Transaction Initiated. Hash: ${tx.hash.substring(0,10)}...`, "text-gray-500");
        createPlanBtn.innerText = "Processing on Blockchain...";
        
        const receipt = await tx.wait();
        
        planCount++;
        document.getElementById('activePlansCount').innerText = planCount;
        
        addLog(`Success! Plan deployed in block ${receipt.blockNumber}.`, "text-green-500 font-bold");
        createPlanBtn.innerText = "Deploy Smart Contract Plan";
        createPlanBtn.disabled = false;
        planAmountInput.value = '';

      } catch (error) {
        addLog("Transaction cancelled or failed.", "text-red-500");
        createPlanBtn.innerText = "Deploy Smart Contract Plan";
        createPlanBtn.disabled = false;
      }
    });
  </script>
