  <script>
    const connectBtn = document.getElementById('connectWalletBtn');
    const walletAddressText = document.getElementById('walletAddress');
    const warningAlert = document.getElementById('walletWarning');
    const createPlanBtn = document.getElementById('createPlanBtn');
    const terminalBody = document.getElementById('terminalBody');
    const planAmountInput = document.getElementById('planAmount');
    const planCycleInput = document.getElementById('planCycle');

    // DAKSH: INPUT YOUR DEPLOYED CONTRACT ADDRESS HERE
    const CONTRACT_ADDRESS = "0x"; 
    const POLYGON_CHAIN_ID = '0x89'; // 137 in Hex

    const CONTRACT_ABI = [
      "function createPlan(uint256 _amount, uint256 _cycle) external returns (uint256)",
      "event PlanCreated(uint256 indexed planId, address indexed merchant, uint256 amount, uint256 cycle)"
    ];

    let userAddress = null;
    let planCount = 0;

    function addLog(text, color = "text-green-400") {
      const p = document.createElement('p');
      p.className = color;
      p.innerText = `> ${text}`;
      terminalBody.appendChild(p);
      terminalBody.scrollTop = terminalBody.scrollHeight;
    }

    connectBtn.addEventListener('click', async () => {
      if (typeof window.ethereum !== 'undefined') {
        try {
          const chainId = await window.ethereum.request({ method: 'eth_chainId' });
          if (chainId !== POLYGON_CHAIN_ID) {
            addLog("Switching to Polygon...", "text-yellow-400");
            try {
              await window.ethereum.request({
                method: 'wallet_switchEthereumChain',
                params: [{ chainId: POLYGON_CHAIN_ID }],
              });
            } catch (err) { addLog("Please switch to Polygon manually.", "text-red-400"); return; }
          }

          const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' });
          userAddress = accounts[0];
          walletAddressText.innerText = userAddress.substring(0, 6) + "..." + userAddress.substring(38);
          connectBtn.classList.replace('bg-gray-900', 'bg-green-600');
          warningAlert.style.display = 'none';
          createPlanBtn.disabled = false;
          createPlanBtn.classList.remove('opacity-50', 'cursor-not-allowed');
          addLog("Connected to Polygon Network.", "text-white font-bold");
        } catch (error) { addLog("Rejected.", "text-red-400"); }
      } else { alert("Install Trust Wallet / MetaMask"); }
    });

    createPlanBtn.addEventListener('click', async () => {
      if (!planAmountInput.value || CONTRACT_ADDRESS === "0x") {
        alert("Enter amount and ensure Contract Address is set!");
        return;
      }

      createPlanBtn.disabled = true;
      createPlanBtn.innerText = "Confirm in Wallet...";
      addLog("Signing transaction...", "text-yellow-400");

      try {
        const provider = new ethers.providers.Web3Provider(window.ethereum);
        const signer = provider.getSigner();
        const contract = new ethers.Contract(CONTRACT_ADDRESS, CONTRACT_ABI, signer);

        const amountInUnits = ethers.utils.parseUnits(planAmountInput.value, 6); 
        const cycleInSeconds = planCycleInput.value === 'Monthly' ? 2592000 : 31536000;

        const tx = await contract.createPlan(amountInUnits, cycleInSeconds);
        addLog(`Tx Sent: ${tx.hash.substring(0,10)}...`, "text-blue-400");
        
        const receipt = await tx.wait();
        planCount++;
        document.getElementById('activePlansCount').innerText = planCount;
        addLog(`Success! Block: ${receipt.blockNumber}`, "text-green-500 font-bold");
        createPlanBtn.innerText = "Deploy Another Plan";
        createPlanBtn.disabled = false;
      } catch (error) {
        addLog("Failed/Cancelled.", "text-red-500");
        createPlanBtn.innerText = "Deploy Plan";
        createPlanBtn.disabled = false;
      }
    });
  </script>
