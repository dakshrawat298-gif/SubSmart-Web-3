  <script>
    // Elements
    const openWalletModalBtn = document.getElementById('openWalletModalBtn');
    const walletAddressText = document.getElementById('walletAddress');
    const warningAlert = document.getElementById('walletWarning');
    const openTxModalBtn = document.getElementById('openTxModalBtn');
    const copyLinkBtn = document.getElementById('copyLinkBtn');
    const terminalBody = document.getElementById('terminalBody');
    const plansTableBody = document.getElementById('plansTableBody');
    const emptyPlansRow = document.getElementById('emptyPlansRow');
    
    // Inputs
    const planNameInput = document.getElementById('planName');
    const planAmountInput = document.getElementById('planAmount');
    const planCycleInput = document.getElementById('planCycle');
    
    // Modals
    const walletModal = document.getElementById('walletModal');
    const cancelWalletBtn = document.getElementById('cancelWalletBtn');
    const confirmWalletBtn = document.getElementById('confirmWalletBtn');
    
    const txModal = document.getElementById('txModal');
    const cancelTxBtn = document.getElementById('cancelTxBtn');
    const confirmTxBtn = document.getElementById('confirmTxBtn');

    let planCount = 0;
    let lastDeployedPlanName = ""; 
    let lastDeployedAmount = "";
    let lastDeployedCycle = "";

    function addLog(text, color = "text-green-400") {
      const p = document.createElement('p');
      p.className = color;
      p.innerText = `> ${text}`;
      terminalBody.appendChild(p);
      terminalBody.scrollTop = terminalBody.scrollHeight;
    }

    // Wallet Modal Logic
    openWalletModalBtn.addEventListener('click', () => {
      walletModal.classList.remove('hidden');
    });

    cancelWalletBtn.addEventListener('click', () => {
      walletModal.classList.add('hidden');
    });

    confirmWalletBtn.addEventListener('click', () => {
      confirmWalletBtn.innerText = "Connecting...";
      
      setTimeout(() => {
        walletModal.classList.add('hidden');
        confirmWalletBtn.innerText = "Sign & Connect"; // reset for next time
        
        // Fake a random wallet address
        const fakeAddress = "0x" + Math.random().toString(16).substr(2, 4) + "..." + Math.random().toString(16).substr(2, 4);
        walletAddressText.innerText = fakeAddress;
        openWalletModalBtn.classList.replace('bg-gray-900', 'bg-green-600');
        warningAlert.style.display = 'none';
        
        openTxModalBtn.disabled = false;
        openTxModalBtn.classList.remove('opacity-50', 'cursor-not-allowed');
        addLog(`Trust Wallet connected: ${fakeAddress}`, "text-white font-bold");
      }, 800);
    });

    // Transaction Modal Logic
    openTxModalBtn.addEventListener('click', () => {
      if (!planNameInput.value || !planAmountInput.value) {
        alert("Please enter the Plan Name and Amount first!");
        return;
      }
      txModal.classList.remove('hidden');
      addLog("Waiting for signature from Trust Wallet...", "text-yellow-400");
    });

    cancelTxBtn.addEventListener('click', () => {
      txModal.classList.add('hidden');
      addLog("Transaction rejected by user.", "text-red-400");
    });

    confirmTxBtn.addEventListener('click', () => {
      confirmTxBtn.innerText = "Processing...";
      
      // Save data before clearing the inputs later
      lastDeployedPlanName = planNameInput.value;
      lastDeployedAmount = planAmountInput.value;
      lastDeployedCycle = planCycleInput.value;
      
      setTimeout(() => {
        txModal.classList.add('hidden');
        confirmTxBtn.innerText = "Confirm"; // reset
        
        openTxModalBtn.disabled = true;
        copyLinkBtn.classList.add('hidden'); // hide link button if deploying another
        const fakeHash = "0x" + Math.random().toString(16).substr(2, 8) + "..." + Math.random().toString(16).substr(2, 4);
        addLog(`Tx Sent: ${fakeHash}`, "text-blue-400");
        openTxModalBtn.innerText = "Processing on Chain...";

        setTimeout(() => {
          planCount++;
          document.getElementById('activePlansCount').innerText = planCount;
          
          const fakeBlock = Math.floor(Math.random() * 100000) + 1900000;
          addLog(`Success! Plan deployed in block ${fakeBlock}.`, "text-green-500 font-bold");
          
          openTxModalBtn.innerText = "Deploy Another Plan";
          openTxModalBtn.disabled = false;
          
          copyLinkBtn.classList.remove('hidden');

          if(emptyPlansRow) emptyPlansRow.remove(); 
          
          const newRow = document.createElement('tr');
          newRow.className = "hover:bg-gray-50 transition-colors";
          newRow.innerHTML = `
            <td class="px-6 py-4 font-bold text-gray-900">${lastDeployedPlanName}</td>
            <td class="px-6 py-4 font-bold text-blue-600">${lastDeployedAmount} USDC</td>
            <td class="px-6 py-4 text-gray-500">${lastDeployedCycle}</td>
            <td class="px-6 py-4"><span class="px-2.5 py-1 bg-green-100 text-green-700 rounded-md text-xs font-bold shadow-sm">Active</span></td>
          `;
          plansTableBody.insertBefore(newRow, plansTableBody.firstChild);

          planNameInput.value = '';
          planAmountInput.value = '';
          planCycleInput.value = 'Monthly';

        }, 2500);

      }, 1000);
    });

    // Copy Link Logic
    copyLinkBtn.addEventListener('click', () => {
      let formattedName = lastDeployedPlanName.trim().toLowerCase().replace(/[^a-z0-9]/g, '-');
      if(!formattedName) formattedName = "plan";
      let fakeUrl = `subsmart.app/pay/${formattedName}`;
      
      copyLinkBtn.innerText = `✅ Copied: ${fakeUrl}`;
      copyLinkBtn.classList.replace('bg-green-500', 'bg-gray-800');
      copyLinkBtn.classList.replace('hover:bg-green-600', 'hover:bg-gray-900');
      addLog(`Payment link for ${lastDeployedPlanName} generated.`, "text-white font-bold");
      
      const realUrlToOpen = window.location.href.replace('dashboard.html', 'checkout.html');
      navigator.clipboard.writeText(realUrlToOpen).catch(err => console.log('Copy failed', err));

      setTimeout(() => {
        copyLinkBtn.innerText = "🔗 Copy Checkout Link";
        copyLinkBtn.classList.replace('bg-gray-800', 'bg-green-500');
        copyLinkBtn.classList.replace('hover:bg-gray-900', 'hover:bg-green-600');
      }, 4000);
    });

    document.addEventListener('keydown', (e) => {
      if(e.key === 's' || e.key === 'S') {
        const subCountEl = document.getElementById('activeSubsCount');
        let currentSub = parseInt(subCountEl.innerText);
        subCountEl.innerText = currentSub + 1;
        addLog("New Subscription Received!", "text-purple-400 font-bold");
      }
    });
  </script>
