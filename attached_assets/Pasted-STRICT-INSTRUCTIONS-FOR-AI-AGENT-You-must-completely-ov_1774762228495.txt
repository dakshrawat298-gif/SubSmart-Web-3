  <script>
    function copyAddress(btn) {
      const address = '0xDA8bab807633f07f013f94D00E6a4B23C9F1e802';
      navigator.clipboard.writeText(address).then(() => {
        const orig = btn.innerHTML;
        btn.textContent = 'Copied!';
        btn.classList.add('bg-green-50', 'border-green-300', 'text-green-700');
        btn.classList.remove('text-blue-600', 'border-blue-200', 'bg-blue-50');
        setTimeout(() => {
          btn.innerHTML = orig;
          btn.classList.remove('bg-green-50', 'border-green-300', 'text-green-700');
          btn.classList.add('text-blue-600', 'border-blue-200', 'bg-blue-50');
        }, 2000);
      });
    }

    // WALLET CONNECTION LOGIC
    document.addEventListener('DOMContentLoaded', () => {
      const connectWalletBtn = document.getElementById('connectWalletBtn');
      
      if (connectWalletBtn) {
        connectWalletBtn.addEventListener('click', async () => {
          if (typeof window.ethereum !== 'undefined') {
            try {
              const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' });
              const account = accounts[0];
              const formattedAddress = `${account.substring(0, 6)}...${account.substring(account.length - 4)}`;
              
              connectWalletBtn.textContent = formattedAddress;
              connectWalletBtn.classList.remove('bg-blue-50', 'text-blue-600', 'hover:bg-blue-100', 'border-blue-200');
              connectWalletBtn.classList.add('bg-gray-100', 'text-gray-900', 'border-gray-200', 'mono');
              connectWalletBtn.disabled = true;
            } catch (error) {
              console.error("Wallet connection failed:", error);
              alert('Connection rejected. Try again.');
            }
          } else {
            alert('Please open this page in a browser with a Web3 wallet (like MetaMask) installed to test connection.');
          }
        });
      }

      // SIMULATION LOGIC
      const simBtn = document.getElementById('simulatePaymentBtn');
      const statusBadge = document.getElementById('statusBadge');
      const statusDot = document.getElementById('statusDot');
      const billingDate = document.getElementById('billingDate');
      const subCard = document.getElementById('subscriptionCard');

      if (simBtn) {
        simBtn.addEventListener('click', () => {
          // Disable button and show loading state
          simBtn.disabled = true;
          simBtn.innerHTML = '<span class="loader"></span> Confirming...';
          simBtn.classList.add('opacity-75', 'cursor-not-allowed');

          // Simulate network delay (2.5 seconds)
          setTimeout(() => {
            // Revert button
            simBtn.innerHTML = 'Payment Successful ✓';
            simBtn.classList.remove('bg-gray-900', 'hover:bg-gray-800');
            simBtn.classList.add('bg-green-600', 'text-white');
            
            // Update UI to Active state
            statusBadge.classList.remove('bg-gray-50', 'border-gray-200', 'text-gray-500');
            statusBadge.classList.add('bg-green-50', 'border-green-200', 'text-green-700');
            statusBadge.innerHTML = '<span class="w-1.5 h-1.5 rounded-full bg-green-500"></span> ACTIVE';
            
            // Set dynamic date (next month)
            const nextMonth = new Date();
            nextMonth.setMonth(nextMonth.getMonth() + 1);
            const dateOptions = { month: 'short', day: 'numeric', year: 'numeric' };
            billingDate.textContent = nextMonth.toLocaleDateString('en-US', dateOptions);

            // Add slight highlight effect to card
            subCard.classList.add('ring-2', 'ring-green-400', 'ring-offset-2');
            setTimeout(() => subCard.classList.remove('ring-2', 'ring-green-400', 'ring-offset-2'), 1500);

          }, 2500);
        });
      }
    });
