  <script>
    // MetaMask Connection Logic
    const connectBtn = document.getElementById('connectWalletBtn');
    const walletAddressText = document.getElementById('walletAddress');
    const warningAlert = document.getElementById('walletWarning');
    const simulateBtn = document.getElementById('simulateBtn');
    const terminalLog = document.getElementById('terminalLog');

    let userAddress = null;

    connectBtn.addEventListener('click', async () => {
      if (typeof window.ethereum !== 'undefined') {
        try {
          // Request account access
          const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' });
          userAddress = accounts[0];
          
          // Truncate address (e.g., 0x1234...ABCD)
          const truncated = userAddress.substring(0, 6) + "..." + userAddress.substring(userAddress.length - 4);
          walletAddressText.innerText = truncated;
          
          // UI Updates
          connectBtn.classList.replace('bg-gray-900', 'bg-green-600');
          connectBtn.classList.replace('hover:bg-black', 'hover:bg-green-700');
          warningAlert.style.display = 'none';
          
          // Enable Simulate Button
          simulateBtn.disabled = false;
          simulateBtn.classList.remove('opacity-50', 'cursor-not-allowed');
          terminalLog.innerText = `> Wallet connected: ${truncated}`;
          terminalLog.classList.remove('animate-pulse');

        } catch (error) {
          console.error("User rejected request");
        }
      } else {
        alert("MetaMask is not installed. Please install it to use this DApp.");
      }
    });

    // Simulate Payment Logic
    simulateBtn.addEventListener('click', () => {
      simulateBtn.disabled = true;
      simulateBtn.innerText = "Processing...";
      terminalLog.innerHTML = `<span class="text-yellow-400">> Requesting USDC approval...</span>`;
      
      setTimeout(() => {
        terminalLog.innerHTML += `<br><span class="text-blue-400">> Smart Contract execution initiated...</span>`;
      }, 1500);

      setTimeout(() => {
        terminalLog.innerHTML += `<br><span class="text-green-400">> Success! Transaction Hash: 0x8f...e3a9</span>`;
        simulateBtn.innerText = "Payment Successful";
        simulateBtn.classList.replace('bg-gray-900', 'bg-green-600');
      }, 3500);
    });
  </script>
